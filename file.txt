pipeline {
    agent any

    environment {
        RECIPIENT_EMAIL = 'zyan0021@gmail.com'
    }

    stages {
        stage('Build') {
            steps {
                script {
                    echo 'Building the code using Maven...'
                    sh 'mvn clean package'
                }
            }
        }

        stage('Unit and Integration Tests') {
            steps {
                script {
                    echo 'Running unit and integration tests...'
                    sh 'mvn test'
                }
            }
            post {
                always {
                    script {
                        def status = currentBuild.currentResult
                        def log = "${env.WORKSPACE}/unit_tests.log"
                        sh "mvn test > ${log}"
                        emailext subject: "Unit and Integration Tests: ${status}",
                                 body: "The Unit and Integration Tests stage has ${status}.",
                                 attachLog: true,
                                 to: "${env.RECIPIENT_EMAIL}"
                    }
                }
            }
        }

        stage('Code Analysis') {
            steps {
                script {
                    echo 'Analyzing code with SonarQube...'
                    sh 'mvn sonar:sonar'
                }
            }
        }

        stage('Security Scan') {
            steps {
                script {
                    echo 'Running security scan with OWASP Dependency-Check...'
                    sh 'mvn dependency-check:check'
                }
            }
            post {
                always {
                    script {
                        def status = currentBuild.currentResult
                        def log = "${env.WORKSPACE}/security_scan.log"
                        sh "mvn dependency-check:check > ${log}"
                        emailext subject: "Security Scan: ${status}",
                                 body: "The Security Scan stage has ${status}.",
                                 attachLog: true,
                                 to: "${env.RECIPIENT_EMAIL}"
                    }
                }
            }
        }

        stage('Deploy to Staging') {
            steps {
                script {
                    echo 'Deploying to staging server...'
                    
                    sh './deploy_to_staging.sh'
                }
            }
        }

        stage('Integration Tests on Staging') {
            steps {
                script {
                    echo 'Running integration tests on staging...'
                   
                    sh './run_selenium_tests.sh'
                }
            }
        }

        stage('Deploy to Production') {
            steps {
                script {
                    echo 'Deploying to production server...'
                   
                    sh './deploy_to_production.sh'
                }
            }
        }
    }
}
